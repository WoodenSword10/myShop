加载数据库

数据库位于sklearn.datasets中

iris鸢尾花数据库

# 导包
from sklearn.datasets import load_iris
# 加载数据库
iris = load_iris()
# 打印数据库全部内容
print(iris)
# 打印特征值
print(iris['data']
# 打印目标值
print(iris['target'])
# 打印数据集描述
print(iris['DESCR'])
# 打印目标名
print(iris['target_names']
# 打印特征名
print(iris['feature_names'])

特征工程

对原始数据进行处理便于后续学习使用

数据集划分

# 导包
from sklearn.datasets import load_iris
from sklearn.model_selection import train_test_split
# 获取数据集
iris = load_iris()
# 数据划分
# 训练集的特征值x_train，测试集的特征值x_test，训练集的目标值y_train，测试集的目标值y_test
x_train, x_test, y_train, y_test = train_test_split(iris['data'], iris['target'], random_state=22)
# 打印结果
print('x_train:', x_train)

归一化：通过对原始数据进行变换把数据映射到（默认）【０－１】之间

公式:
x` = (x-min)/(max-min)
x`` = x` * (mx-mi) + mi
缺点：最大值最小值非常容易受异常点的影响，鲁棒性较差，只适合传统精确小数据场景

# 导包
from sklearn.datasets import load_iris
from sklearn.preprocessing import MinMaxScaler
# 数据准备
iris = load_iris()
# 实例化转换器类
transfer= MinMaxScaler()
# 调用fit_transform
data = transfer.fit_transform(iris['data'])

标准化

公式：
x` = (x-mean)/a
mean为平均值，a为标准差
作用于每一列
在已有样本足够多的情况下比较稳定，适合现代嘈杂大数据场景

# 导包
from sklearn.preprocessing import StandardScaler
from sklearn.datasets import load_iris
# 获取数据
iris = load_iris()
# 实例化转换器类
transfer = StandardScaler()
# 调用fit_transform
data = transfer.fit_transform(iris['data'])
# 打印结果
print('标准化结果：', data)
print('每一列特征的平均值：', transfer.mean_)
print('每一列特征的方差：', transfer.val_)
